                    const now = new Date();

                        setTimeLeft({

            // 通用 Gemini API 呼叫函式

            const callGeminiAPI = async (payload) => {

                if (!apiKey) {

                    setIsSettingsOpen(true);

                    throw new Error("請先設定 Gemini API Key。");

                }

                const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

                const delays = [1000, 2000, 4000, 8000, 16000];

                for (let i = 0; i < 5; i++) {

                    try {

                        const response = await fetch(url, {

                            method: "POST",

                            headers: { "Content-Type": "application/json" },

                            body: JSON.stringify(payload)

                        });

                        if (response.status === 403 || response.status === 401) {

                            setIsSettingsOpen(true);

                            throw new Error("API Key 無效或無權限，請重新檢查。");

                        }

                        if (!response.ok) throw new Error(`API Error: ${response.status}`);

                        const data = await response.json();

                        const resultText = data.candidates?.[0]?.content?.parts?.[0]?.text;

                        if (!resultText) throw new Error("無效的回傳格式");

                        return resultText;

                    } catch (err) {

                        if (i === 4 || err.message.includes("API Key")) throw err;

                        await new Promise(res => setTimeout(res, delays[i]));

                    }

                }

            };


            const generateExam = async () => {

                if (!apiKey) {

                    setIsSettingsOpen(true);

                setIsLoading(true);

                const payload = {

                    generationConfig: {

                        responseMimeType: "application/json",

                        responseSchema: {

                            type: "OBJECT",

                            properties: {

                                mcqs: { type: "ARRAY", items: { type: "OBJECT", properties: { question: { type: "STRING" }, options: { type: "ARRAY", items: { type: "STRING" } }, correctAnswerIndex: { type: "INTEGER" }, explanation: { type: "STRING" } }, required: ["question", "options", "correctAnswerIndex", "explanation"] } },

                                comprehensive: { type: "OBJECT", properties: { scenario: { type: "STRING" }, subQuestion1: { type: "STRING" }, subQuestion2: { type: "STRING" }, answer1: { type: "STRING" }, answer2: { type: "STRING" } }, required: ["scenario", "subQuestion1", "subQuestion2", "answer1", "answer2"] }

                            },

                            required: ["mcqs", "comprehensive"]

                        }

                    }

                };


                try {

                    const resultText = await callGeminiAPI(payload);

            const handleGenerateMnemonic = async (qIndex) => {

                setGeneratingMnemonics(prev => ({ ...prev, [qIndex]: true }));

                const q = examData.mcqs[qIndex];

                const currentSubject = SUBJECTS[selectedSubjectId];

                const payload = { contents: [{ parts: [{ text: `這是一題台灣教檢「${currentSubject.name}」的考題：\n題目：${q.question}\n解答與解析：${q.explanation}\n\n請設計一個「✨ 簡單直白的記憶口訣」或「生動的聯想記憶法」，50-100 字內幫助考生記憶。` }] }] };

                try {

                    const resultText = await callGeminiAPI(payload);

                    setMnemonics(prev => ({ ...prev, [qIndex]: resultText }));

                } catch (err) { console.error(err); } finally { setGeneratingMnemonics(prev => ({ ...prev, [qIndex]: false })); }

            };


            const handleGradeComp = async () => {

                if (!compUserInputs.q1.trim() && !compUserInputs.q2.trim()) return alert("請先填寫作答內容。");

                setIsGrading(true);

                const comp = examData.comprehensive;

                const gradingPrompt = `你是一位嚴格的台灣教檢閱卷委員。請針對「子題一」與「子題二」分別評分（滿分10分）。\n情境：${comp.scenario}\n子題一題目：${comp.subQuestion1}\n考生作答：${compUserInputs.q1}\n標準答案：${comp.answer1}\n\n子題二題目：${comp.subQuestion2}\n考生作答：${compUserInputs.q2}\n標準答案：${comp.answer2}\n請以 JSON 回傳分開的評分 q1, q2 (score, praise, critique, suggestion)。`;

                const payload = { contents: [{ parts: [{ text: gradingPrompt }] }], generationConfig: { responseMimeType: "application/json" } };

                try {

                    const resultText = await callGeminiAPI(payload);

                    setCompGradeResult(JSON.parse(resultText));

                } catch (err) { console.error(err); } finally { setIsGrading(false); }

            };


            const handleGenerateStudyGuide = async () => {

                setIsGeneratingGuide(true);
